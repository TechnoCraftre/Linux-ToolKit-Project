# Create directories
sudo mkdir -p /opt/linux_toolkit/logs
sudo mkdir -p /opt/linux_toolkit/bin

# Navigate into project directory
cd /opt/linux_toolkit

# Create a log file
touch logs/app.log

# Copy file
cp logs/app.log logs/app_copy.log

# Rename (move)
mv logs/app_copy.log logs/app.1

# Change permissions
chmod 640 logs/app.log

# Change ownership to admin1
sudo chown admin1:admin1 logs/app.log

# Find log files
find /opt/linux_toolkit -name "*.log"
